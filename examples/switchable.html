Slide - 卡盘轮播

seajs.config({
        alias: {
            'tabs-src': '../src/tabs',
            'slide-src': '../src/slide'
        }
    });

    var T, $, tabs, slide;

    seajs.use(['jquery', 'tabs-src', 'slide-src', 'easing'], function(jQuery, Tabs, Slide) {
        T = Tabs;
        $ = jQuery;

        tabs = new Tabs({
            element: '#demo1',
            activeIndex: 1,
            effect: 'fade'
        });

        slide = new Slide({
            element: '#demo2',
            effect: 'scrolly',
            interval: 2000
        });

    });
